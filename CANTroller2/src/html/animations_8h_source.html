149 bool mainfunc(void) {
150 bool new_round = false;
151 static constexpr float e = 0.999; // Coefficient of friction
152 sec = lgfx::millis() / ball_create_rate;
153 if (psec != sec) {
154 psec = sec;
155 myfps = frame_count;
156 frame_count = 0;
157 if (++ball_count >= ball_thismax) {
158 new_round = true;
159 ball_count = 1;
160 }
161 new_ball(ball_count - 1);
162 }
163 frame_count++;
164 _loop_count++;
165 ball_info_t *a, *b, *balls;
166 int32_t rr, len, vx2vy2;
167 float vx, vy, distance, t;
168 size_t f = _loop_count & 1;
169 balls = a = &_balls[f][0];
170 b = &_balls[!f][0];
171 memcpy(a, b, sizeof(ball_info_t) * ball_count);
172 for (int i = 0; i < ball_count + touchnow; i++) {
173 a = (i == ball_count) ? &touchball : &balls[i];
174 if (i < ball_count) {
175 a->dy += ball_gravity; // gravity
176 a->x += a->dx;
177 if (a->x < a->r) {
178 a->x = a->r;
179 if (a->dx < 0) a->dx = -a->dx * e;
180 }
181 else if (a->x >= _width - a->r) {
182 a->x = _width - a->r - 1;
183 if (a->dx > 0) a->dx = -a->dx * e;
184 }
185 a->y += a->dy;
186 if (a->y < a->r) {
187 a->y = a->r;
188 if (a->dy < 0) a->dy = -a->dy * e;
189 }
190 else if (a->y >= _height - a->r) {
191 a->y = _height - a->r - 1;
192 if (a->dy > 0) a->dy = -a->dy * e;
193 }
194 }
195 for (int j = i + 1; j < ball_count + touchnow; j++) {
196 b = (j == ball_count) ? &touchball : &balls[j];
197 rr = a->r + b->r;
198 vx = a->x - b->x;
199 if (abs(vx) > rr) continue;
200 vy = a->y - b->y;
201 if (abs(vy) > rr) continue;
202 len = sqrt(vx * vx + vy * vy);
203 if (len >= rr) continue;
204 if (len == 0.0) continue;
205 distance = (rr - len) >> 1;
206 vx *= distance / len;
207 vy *= distance / len;
208 a->x += vx;
209 if (j < ball_count) b->x -= vx;
210 vx = b->x - a->x;
211 a->y += vy;
212 if (j < ball_count) b->y -= vy;
213 vy = b->y - a->y;
214 vx2vy2 = vx * vx + vy * vy;
215 t = -(vx * a->dx + vy * a->dy) / vx2vy2;
216 float arx = a->dx + vx * t;
217 float ary = a->dy + vy * t;
218 t = -(-vy * a->dx + vx * a->dy) / vx2vy2;
219 float amx = a->dx - vy * t;
220 float amy = a->dy + vx * t;
221 t = -(vx * b->dx + vy * b->dy) / vx2vy2;
222 float brx = b->dx + vx * t;
223 float bry = b->dy + vy * t;
224 t = -(-vy * b->dx + vx * b->dy) / vx2vy2;
225 float bmx = b->dx - vy * t;
226 float bmy = b->dy + vx * t;
227 float adx = (a->m * amx + b->m * bmx + bmx * e * b->m - amx * e * b->m) / (a->m + b->m);
228 float bdx = -e * (bmx - amx) + adx;
229 float ady = (a->m * amy + b->m * bmy + bmy * e * b->m - amy * e * b->m) / (a->m + b->m);
230 float bdy = -e * (bmy - amy) + ady;
231 a->dx = roundf(adx + arx);
232 a->dy = roundf(ady + ary);
233 if (j < ball_count) {
234 b->dx = roundf(bdx + brx);
235 b->dy = roundf(bdy + bry);
236 }
237 }
238 }
239 _myfps = myfps;
240 _ball_count = ball_count;
241 return new_round;

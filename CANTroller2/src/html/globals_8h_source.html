231// Exponential Moving Average filter : smooth out noise on inputs. 0 < alpha < 1 where lower = smoother and higher = more responsive
232// pass in a fresh raw value, address of filtered value, and alpha factor, filtered value will get updated
233float ema_filt(float _raw, float _filt, float _alpha) {
234 return (_alpha * _raw) + ((1 - _alpha) * _filt);
235}
236template<typename RAW_T, typename FILT_T>
237void ema_filt(RAW_T _raw, FILT_T* _filt, float _alpha) {
238 float _raw_f = static_cast<float>(_raw);
239 float _filt_f = static_cast<float>(*_filt);
240 *_filt = static_cast<FILT_T>(ema_filt(_raw_f, _filt_f, _alpha));
241}
242// functions for changing values while respecting min and max constraints. used in tuner ui
243template <typename T>
244T adj_val(T variable, T modify, T low_limit, T high_limit) {
245 T oldval = variable;
246 if (std::is_same<T, int32_t>::value) variable += (T)modify;
247 else if (std::is_same<T, float>::value) variable += (T)(modify * tuning_rate_pcps * (high_limit-low_limit) * loop_avg_us / (1000000 * 100.0));
248 return variable < low_limit ? low_limit : (variable > high_limit ? high_limit : variable);
249}
250bool adj_val(int32_t *variable, int32_t modify, int32_t low_limit, int32_t high_limit) { // sets an int reference to new val constrained to given range
251 int32_t oldval = *variable;
252 *variable = adj_val(*variable, modify, low_limit, high_limit);
253 return (*variable != oldval);
254}

264T hsv_to_rgb(uint16_t hue, uint8_t sat = 255, uint8_t val = 255) {
265 uint8_t rgb[3] = { 0, 0, 0 }; // [r,g,b];

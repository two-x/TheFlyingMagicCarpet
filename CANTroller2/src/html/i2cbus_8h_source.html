140float SparkFun_MicroPressure::readPressure(Pressure_Units units, bool noblock) {
141 if (run_preamble) {
142 _i2cPort->beginTransmission(_address);
143 _i2cPort->write((uint8_t)0xAA);
144 _i2cPort->write((uint8_t)0x00);
145 _i2cPort->write((uint8_t)0x00);
146 _i2cPort->endTransmission();
147 }
148 run_preamble = false;
149 if (_eoc != -1) { // Use GPIO pin if defined
150 while (!digitalRead(_eoc)) {
151 if (noblock) return NAN;
152 delay(1);
153 }
154 }
155 else { // Check status byte if GPIO is not defined
156 _status = readStatus();
157 while((_status&BUSY_FLAG) && (_status!=0xFF)) {
158 if (noblock) return NAN;
159 delay(1);
160 _status = readStatus();
161 }
162 }
163 run_preamble = true;
164 _i2cPort->requestFrom(_address,4);
165 _status = _i2cPort->read();
166 if((_status & INTEGRITY_FLAG) || (_status & MATH_SAT_FLAG)) return NAN; // check memory integrity and math saturation bit
167 uint32_t reading = 0;
168 for(uint8_t i=0;i<3;i++) { // read 24-bit pressure
169 reading |= _i2cPort->read();
170 if(i != 2) reading = reading<<8;
171 }
172 pressure = (reading - OUTPUT_MIN) * (_maxPsi - _minPsi);
173 pressure = (pressure / (OUTPUT_MAX - OUTPUT_MIN)) + _minPsi;
174 if(units == PA) pressure *= 6894.7573; //Pa (Pascal)
175 else if(units == KPA) pressure *= 6.89476; //kPa (kilopascal)
176 else if(units == TORR) pressure *= 51.7149; //torr (mmHg)
177 else if(units == INHG) pressure *= 2.03602; //inHg (inch of mercury)
178 else if(units == ATM) pressure *= 0.06805; //atm (atmosphere)
179 else if(units == BAR) pressure *= 0.06895; //bar
180 return pressure;
181}

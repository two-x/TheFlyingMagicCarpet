38 _suppress_click = true; // Prevents the switch release after a long press from causing a short press
39 }
40 now = true; // Remember a press is in effect
41 }
42 else { // if encoder sw is not being pressed
43 if (now) {

46 }
47 _timer_active = false; // Allows detection of next long press event
48 now = false; // Remember press is not in effect

50 }

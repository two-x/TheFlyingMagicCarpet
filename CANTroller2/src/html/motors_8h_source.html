64 // This function should be called every time "void loop()" executes. The function will decide whether a new
65 // PID output needs to be computed. Returns true when the output is computed, false when nothing has been done.
66 float compute() {
67 uint32_t now = micros();
68 uint32_t timechange = (now - lasttime);
69 if (_mode == ctrl::automatic && timechange < sampletime) return _output; // If class is handling the timing and this time was a nop
70
71 float in = *myin;
72 float din = in - lastin;
73 if (_dir == cdir::reverse) din = -din;
74
75 _err = _target - in;
76 if (_dir == cdir::reverse) _err = -_err;
77 float derr = _err - lasterr;
78
79 float peterm = _kp * _err;
80 float pmterm = _kp * din;
81 if (_pmode == pmod::onerr) pmterm = 0;
82 else if (_pmode == pmod::onmeas) peterm = 0;
83 else { //onerrmeas
84 peterm *= 0.5f;
85 pmterm *= 0.5f;
86 }
87 _pterm = peterm - pmterm;
88 _iterm = _ki * _err;
89 if (_dmode == dmod::onerr) _dterm = _kd * derr;
90 else _dterm = -_kd * din; // onmeas
91
92 if (_awmode == awmod::cond || _awmode == awmod::roundcond) { // condition anti-windup (default)
93 bool aw = false;
94 float _itermout = (peterm - pmterm) + _ki * (_iterm + _err);
95 if (_itermout > *_outmax && derr > 0) aw = true;
96 else if (_itermout < *_outmin && derr < 0) aw = true;
97 if (aw && _ki) _iterm = constrain(_itermout, -(*_outmax), *_outmax);
98 }
99 else if ((_awmode == awmod::round || _awmode == awmod::roundcond) && _err < 0.001 && _err > -0.001) {
100 _err = 0.0;
101 if (_centmode == centmod::on || _centmode == centmod::strict) _outsum = _cent;
102 }
103 if (_centmode == centmod::strict && _err * lasterr < 0) _outsum = _cent; // Recenters any old integral when error crosses zero
104
105 _outsum += _iterm - pmterm; // by default, compute output as per PID_v1 // include integral amount and pmterm
106 if (_awmode != awmod::off) _outsum = constrain(_outsum, *_outmin, *_outmax); // Clamp
107
108 _output = constrain(_outsum + peterm + _dterm, *_outmin, *_outmax); // include _dterm, clamp and drive output
109
110 lasterr = _err;
111 lastin = in;
112 lasttime = now;
113 return _output;
114 }

124 _kp = a_kp;
125 _ki = a_ki * sampletime_sec;
126 _kd = a_kd / sampletime_sec;
127 }

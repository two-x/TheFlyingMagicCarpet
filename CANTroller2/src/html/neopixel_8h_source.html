196void NeopixelStrip::heartbeat_update() {
197 if (!neo_heartbeat) return;
198 if (neoHeartbeatTimer.expired()) {
199 heartbeat_pulse = !heartbeat_pulse;
200 ++heartbeat_state %= arraysize(heartbeat_ekg_us);
201 neoHeartbeatTimer.set(heartbeat_ekg_us[heartbeat_state]);
202 if (heartbeat_pulse) heartbeat_brightness = heartbright;
203 else neoFadeTimer.reset();
204 }
205 else if (!heartbeat_pulse) {
206 heartbeat_brightness_last = heartbeat_brightness;
207 if (neoFadeTimer.expired()) heartbeat_brightness = brightlev[context][B_MIN];
208 else heartbeat_brightness = (int8_t)(heartlobright + std::max((double)0, (float)(heartbright - heartlobright) * (1.0 - ((heartbeat_state == 1) ? 1.5 : 1.0) * (float)neoFadeTimer.elapsed() / (float)neo_fade_timeout_us)));
209 if (heartbeat_brightness > heartbeat_brightness_last) heartbeat_brightness = heartbeat_brightness_last;
210 }
211 if (heartcolor_change || heartbeat_brightness != neobright_last) {
212 heartbeatNow = dimmer(heartbeatColor, heartbeat_brightness); // heartbeatNow = dimmer(desaturate(heartbeatColor, desat_of_ten), heartbeat_brightness);
213 heartcolor_change = false;
214 neobright_last = heartbeat_brightness;

242// setflash() : Call this to add a blink sequence to one of the idiot lights which will repeat indefinitely in [up to 6.4 sec] cycles
243// _idiot = which idiot light
244// count = number of blinks per cycle. Use 0 to cancel a previously applied blink pattern
245// pulseh = high pulse width of each blink, in increments of 50 ms (of which there are max 128 per cycle)
246// pulsel = low pulse width of each blink, in increments of 50 ms (of which there are max 128 per cycle)
247// onbrit = percent brightness to apply during high pulses (use -1 to avoid)
248// color = alternate color to apply during high pulses (use -1 to avoid)
249void NeopixelStrip::setflash(uint _idiot, uint8_t count, uint8_t pulseh, uint8_t pulsel, int32_t onbrit, int32_t color) {
250 fset[_idiot][fcount] = count;
251 fset[_idiot][fpulseh] = std::max((int)pulseh, 1);
252 fset[_idiot][fpulsel] = std::max((int)pulsel, 1);
253 fset[_idiot][fonbrit] = (onbrit == -1) ? hibright : (uint8_t)onbrit;
254 cidiot[_idiot][cflash] = dimmer((color == -1) ? cidiot[_idiot][cnormal] : color_to_neo((uint32_t)color), hibright);
255 cidiot[_idiot][cflash] = desaturate(dimmer(cidiot[_idiot][cflash], fset[_idiot][fonbrit]), desat_of_ten);
256 for (uint8_t pg = 0; pg < fevpages; pg++) fevents[_idiot][pg] = 0;
257 uint8_t filled = 0;
258 uint8_t lstop;
259 uint8_t patternlen = fset[_idiot][fcount] * (fset[_idiot][fpulseh] + fset[_idiot][fpulsel]);
260 uint8_t reps = 1 + (patternlen < fevresolution / 3); // For shorter flash patterns repeat them multiple times in each cycle
261 for (uint8_t rep = 1; rep <= reps; rep++) {
262 lstop = std::min((int)fevresolution, filled + patternlen);
263 while (filled < lstop) {
264 for (uint8_t hbit = 0; hbit < fset[_idiot][fpulseh]; hbit++)
265 if (filled < lstop) fevpush(_idiot, filled++, 1);
266 for (uint8_t lbit = 0; lbit < fset[_idiot][fpulsel]; lbit++)
267 if (filled < lstop) fevpush(_idiot, filled++, 0);
268 }
269 filled = fevresolution / reps;
270 }

287void NeopixelStrip::update_idiot(uint32_t _idiot) {
288 cidiot[_idiot][clast] = cidiot[_idiot][cnow];
289 neorgb_t newnow = (fset[_idiot][onoff]) ? cidiot[_idiot][con] : cidiot[_idiot][coff];

293void NeopixelStrip::update(int16_t heart_color) {
294 set_heartcolor(heart_color);
295 heartbeat_update(); // Update our beating heart
296 nowtime_us = (uint32_t)flashtimer.elapsed();
297 nowepoch = nowtime_us / fquantum_us;
298 for (int32_t i=0; i<idiotcount; i++) {
299 if (!syspower) cidiot[i][cnow] = neorgb_t(0);
300 else update_idiot(i);
301 }
302 refresh();
303 flashtimer.expireset();
304}

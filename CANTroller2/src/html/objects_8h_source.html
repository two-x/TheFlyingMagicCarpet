106// Calculates massairflow in g/s using values passed in if present, otherwise it reads fresh values
107float massairflow(float _map = NAN, float _airvelo = NAN, float _ambient = NAN) { // mdot (kg/s) = density (kg/m3) * v (m/s) * A (m2) . And density = P/RT. So, mdot = v * A * P / (R * T) in kg/s
108 static float maf_map_last;
109 static float maf_velo_last;
110 float temp = _ambient;
111 float new_velo = airvelo.human();
112 float new_map = mapsens.human();
113 if (std::isnan(_ambient)) {
114 if (new_velo == maf_velo_last && new_map == maf_map_last) return maf_gps; // if no new sensor readings, don't recalculate the same value
115 temp = tempsens.val(loc::AMBIENT);
116 if (std::isnan(temp) && running_on_devboard) temp = tempsens.val(loc::ENGINE);
117 if (std::isnan(temp)) return -1; // Avoid crashing due to trying to read absent sensor
118 }
119 maf_velo_last = new_velo;
120 maf_map_last = new_map;
121 float T = 0.556 * (temp - 32.0) + 273.15; // in K. This converts from degF to K
122 float R = 287.1; // R (for air) in J/(kg·K) ( equivalent to 8.314 J/(mol·K) ) 1 J = 1 kg*m2/s2
123 float v = 0.447 * (std::isnan(_airvelo) ? new_velo : _airvelo); // in m/s 1609.34 m/mi * 1/3600 hr/s = 0.447
124 float Ain2 = 3.1415926; // in in2 1.0^2 in2 * pi // will still need to divide by 1550 in2/m2
125 float P = 101325.0 * (std::isnan(_map) ? new_map : _map); // in Pa 101325 Pa/atm 1 Pa = 1 J/m3
126 float maf = v * Ain2 * P * 1000.0 / (R * T * 1550); // mass air flow in grams per second (g/s) (1000 g/kg * m/s * in2 * J/m3) / (J/(kg*K) * K * 1550 in2/m2) = g/s
127 if (std::abs(maf) < 0.001) maf = 0;
128 return maf;
129}

198 if (panic_req == REQ_TOG) panic_req = !panicstop;
199 if (ign_req == REQ_TOG) ign_req = !signal;
200 // else if (request == signal) request = REQ_NA; // With this line, it ignores requests to go to state it's already in, i.e. won't do unnecessary pin write
201 if (speedo.car_stopped() || panicTimer.expired()) panic_req = REQ_OFF; // Cancel panic stop if car is stopped
202 if (!speedo.car_stopped() && (runmode == FLY || runmode == CRUISE || runmode == HOLD)) {
203 if (signal && ign_req == REQ_OFF) panic_req = REQ_ON; // ignition cut while driving causes panic stop

210 if (panicstop) panicTimer.reset();
211 }
212 }
213 paniclast = panicstop;
214 if (panicstop) ign_req = REQ_OFF; // panic stop causes ignition cut
215 if (ign_req != REQ_NA && runmode != ASLEEP) {
216 signal = (bool)ign_req;
217 write_pin(pin, signal); // turn car off or on (ign output is active high), ensuring to never turn on the ignition while panicking

224class Starter {
225 private:
226 uint32_t pushbrake_timeout = 6000000;

237 Serial.printf("Starter.. output-only supported\n");

241 void update() { // starter drive handler logic. Outside code interacts with handler by calling request(XX) = REQ_OFF, REQ_ON, or REQ_TOG
242 // if (now_req != NA) Serial.printf("m:%d r:%d\n", motor, now_req);
243 if (now_req == REQ_TOG) now_req = !motor; // translate a toggle request to a drive request opposite to the current drive state
244 req_active = (now_req != REQ_NA); // for display
245 if (motor && ((now_req == REQ_OFF) || starterTimer.expired())) { // if we're driving the motor but need to stop or in the process of stopping
246 motor = LOW; // we will turn it off
247 write_pin (pin, motor); // begin driving the pin low voltage
248 if (gas.motormode == Starting) gas.setmode(lastgasmode); // put the throttle back to doing whatever it was doing before
249 now_req = REQ_NA;
250 return; // ditch out, leaving the motor-off request intact. we'll check on the timer next time
251 } // now, we have stopped driving the starter if we were supposed to stop
252 // if (sim.simulating(sens::starter)) motor = pin_outputting; // if simulating starter, there's no external influence
253 if (motor || now_req != REQ_ON) { // if starter is already being driven, or we aren't being tasked to drive it
254 now_req = REQ_NA; // cancel any requests
255 return; // and ditch
256 } // from here on, we can assume the starter is off and we are supposed to turn it on
257 if (brake.autoholding || !brake_before_starting) { // if the brake is being held down, or if we don't care whether it is
258 lastgasmode = gas.motormode; // remember incumbent gas setting
259 gas.setmode(Starting); // give it some gas
260 starterTimer.set((int64_t)run_timeout); // if left on the starter will turn off automatically after X seconds
261 motor = HIGH; // ensure starter variable always reflects the starter status regardless who is driving it
262 write_pin(pin, motor); // and start the motor
263 now_req = REQ_NA; // we have serviced starter-on request, so cancel it
264 return; // if the brake was right we have started driving the starter
265 } // from here on, we can assume the brake isn't being held on, which is in the way of our task to begin driving the starter
266 if (brake.motormode != AutoHold) { // if we haven't yet told the brake to hold down
267 lastbrakemode = brake.motormode; // remember incumbent brake setting
268 brake.setmode(AutoHold); // tell the brake to hold
269 starterTimer.set((int64_t)pushbrake_timeout); // start a timer to time box that action
270 return; // we told the brake to hold down, leaving the request to turn the starter on intact, so we'll be back to check
271 } // at this point the brake has been told to hold but isn't holding yet
272 if (starterTimer.expired()) { // if we've waited long enough for the damn brake
273 if (brake.motormode == AutoHold) brake.setmode(lastbrakemode); // put the brake back to doing whatever it was doing before, unless it's already been changed
274 now_req = REQ_NA; // cancel the starter-on request, we can't drive the starter cuz the car might lurch forward
275 } // otherwise we're still waiting for the brake to push. the starter turn-on request remains intact
276 }
277 // src source() { return pin_outputting ? src::CALC : src::PIN; }
278};

281class FuelPump { // drives power to the fuel pump when the engine is turning
282 public:
283 float off_v = 0.0;
284 float on_min_v = 8.0;
285 float on_max_v = 12.0;
286 float volts = 0.0;
287 float turnon_rpm = 50.0;
288 float duty, pwm_period = 25000; // used for software pwm timing
289 int adc = 0;
290 bool status = LOW, status_inverse = HIGH, pump_last = LOW, sw_pwm_out_now = LOW;
291 private:
292 bool variable_speed_output = false; // this interferes with the gas servo pwm when enabled
293 bool use_software_pwm = true; // avoid using hardware resources for variable output, we can fake it with a timer
294 Timer fuelTimer;
295 int timeout = 25000; // not tunable
296 int pin, ledc_channel, pwm_frequency = 42, pwm_resolution = 8; // for if hardware pwm is used
297 void writepin() {
298 if (variable_speed_output) {
299 if (use_software_pwm) {
300 if (duty > 99.5) write_pin(pin, HIGH);
301 else if (duty < 0.5) write_pin(pin, LOW);
302 else if (fuelTimer.expired()) {
303 sw_pwm_out_now = !sw_pwm_out_now;
304 timeout = (int)(duty * pwm_period / 100.0);
305 if (!sw_pwm_out_now) timeout = (int)pwm_period - timeout;
306 write_pin(pin, sw_pwm_out_now);
307 fuelTimer.set(timeout);
308 }
309 }
310 else ledcWrite(ledc_channel, adc);
311 }
312 else write_pin(pin, status);
313 }
314 public:
315 FuelPump(int _pin) : pin(_pin) {}
316 void update() {
317 if (!fuelpump_supported || !captouch) return;
318 float tachnow = tach.filt();
319 pump_last = status;
320 if (starter.motor || (ignition.signal && (tachnow >= turnon_rpm))) {
321 volts = map(gas.pc[OUT], gas.pc[OPMIN], gas.pc[OPMAX], on_min_v, on_max_v);
322 volts = constrain(volts, on_min_v, on_max_v);
323 adc = map((int)volts, 0, (int)on_max_v, 0, 255);
324 duty = 100.0 * volts / on_max_v;
325 status = HIGH;
326 }
327 else {
328 volts = off_v;
329 adc = 0;
330 duty = 0.0;
331 status = LOW;
332 }
333 status_inverse = !status; // for idiot light
334 writepin();
335 }
336 void setup() {
337 if (!fuelpump_supported || !captouch) return; // if resistive touchscreen, then pin is needed for chip select
338 Serial.printf("Fuel pump.. ");
339 if (variable_speed_output) {
340 if (use_software_pwm) {
341 set_pin(pin, OUTPUT); // initialize_pin
342 fuelTimer.set(timeout); // start the timer in case we are doing software pwm
343 Serial.printf("using software pwm\n");
344 }
345 else {
346 int ledc_channel = analogGetChannel(pin);
347 if (ledcSetup(ledc_channel, pwm_frequency, pwm_resolution) == 0) Serial.printf("failed to configure ");
348 else {
349 Serial.printf("using ");
350 ledcAttachPin(pin, ledc_channel);
351 }
352 Serial.printf("ledc ch %d, %d bit at %d Hz\n", ledc_channel, pwm_resolution, pwm_frequency);
353 }
354 }
355 else {
356 set_pin(pin, OUTPUT); // initialize_pin
357 Serial.printf("using digital drive\n");
358 }
359 }
